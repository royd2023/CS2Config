    import type { CrosshairState } from '@/types/crosshair';

    export function crosshairCodeGen(state: CrosshairState): string {
      // Map CrosshairState nested fields → flat Crosshair shape expected by csgo-sharecode
      // Inspect node_modules/csgo-sharecode/dist/index.d.ts for exact property names
      const crosshair = {
        style: state.style,
        // Map line fields
        length: state.line.length,
        thickness: state.line.thickness,
        gap: state.line.gap,
        // Map color fields (library uses red/green/blue/alpha or r/g/b/a — check types)
        // Adjust property names to match the library's Crosshair interface exactly
        red: state.color.r,
        green: state.color.g,
        blue: state.color.b,
        alpha: state.color.a,
        // Map boolean/optional fields
        outline: state.outline.enabled ? state.outline.thickness : 0,
        centerDotSize: state.dot.enabled ? 1 : 0,  // or per library convention
        tStyle: state.tStyle,
        followRecoil: state.followRecoil,
        // Fill remaining required library fields with sensible defaults
        // (splitDistance, fixedCrosshairGap, innerSplitAlpha, outerSplitAlpha, splitSizeRatio,
        //  deployedWeaponGapEnabled, alphaEnabled, color — check types for required vs optional)
      };

      return encodeCrosshair(crosshair as Parameters<typeof encodeCrosshair>[0]);
    }

    ```typescript
    // src/lib/cfgGenerator.ts
    // Pure function — no React imports, no side effects
    import type { ConfigState, BindAction } from '@/types/config';
    import { DEPRECATED_CS2_COMMANDS } from '@/lib/constants/deprecatedCmds';

    // CS2 bind action → console command mapping
    const BIND_COMMAND: Record<BindAction, string> = {
      jump: '+jump',
      duck: '+duck',
      attack: '+attack',
      attack2: '+attack2',
      use: '+use',
      reload: '+reload',
      inspect: '+inspect_weapon',
      scoreboard: '+showscores',
      voiceChat: '+voicerecord',
      buyMenu: 'buymenu',
      grenadeFlash: 'slot7',
      grenadeMolotov: 'slot10',
      grenadeSmoke: 'slot8',
      grenadeHE: 'slot6',
      grenadeDecoy: 'slot9',
    };

    export function cfgGenerator(state: ConfigState): string {
      const lines: string[] = [
        '// Generated by CS2 Setup Builder',
        '// autoexec.cfg',
        '',
        '// Mouse',
        `sensitivity ${state.mouse.sensitivity}`,
        '',
        '// Audio',
        `volume ${state.audio.masterVolume}`,
        `voice_scale ${state.audio.voiceVolume}`,
        `voice_enable ${state.audio.muteEnemyVoice ? '0' : '1'}`,
        '',
        '// Key Binds',
      ];

      // Emit bind commands
      for (const [action, key] of Object.entries(state.binds) as [BindAction, string][]) {
        const cmd = BIND_COMMAND[action];
        if (cmd) {
          lines.push(`bind "${key}" "${cmd}"`);
        }
      }

      // Add host_writeconfig to persist settings
      lines.push('');
      lines.push('host_writeconfig');

      const output = lines.join('\n');

      // Runtime safety check — ensure no deprecated commands slipped in
      // (this is a development guard; the unit test provides the real assertion)
      if (process.env.NODE_ENV !== 'production') {
        for (const cmd of DEPRECATED_CS2_COMMANDS) {
          if (output.includes(cmd)) {
            console.error(`cfgGenerator: deprecated command found in output: ${cmd}`);
          }
        }
      }

      return output;
    }
    ```

    NOTE: Launch options are NOT written to autoexec.cfg — they go in Steam launch options. The cfgGenerator only writes cfg commands. If Phase 3 changes this, the generator will be extended then.

    // src/tests/cfgGenerator.test.ts
    import { cfgGenerator } from '@/lib/cfgGenerator';
    import { DEPRECATED_CS2_COMMANDS } from '@/lib/constants/deprecatedCmds';
    import { DEFAULT_CONFIG } from '@/stores/configStore';

    describe('cfgGenerator', () => {
      describe('deprecated command safety', () => {
        it('never outputs deprecated CS2 commands', () => {
          const cfg = cfgGenerator(DEFAULT_CONFIG);
          for (const cmd of DEPRECATED_CS2_COMMANDS) {
            expect(cfg).not.toContain(cmd);
          }
        });

        it('never outputs cl_updaterate regardless of config', () => {
          const cfg = cfgGenerator(DEFAULT_CONFIG);
          expect(cfg).not.toContain('cl_updaterate');
        });

        it('never outputs cl_interp regardless of config', () => {
          const cfg = cfgGenerator(DEFAULT_CONFIG);
          expect(cfg).not.toContain('cl_interp');
        });
      });

      describe('valid command output', () => {
        it('includes sensitivity command', () => {
          const cfg = cfgGenerator(DEFAULT_CONFIG);
          expect(cfg).toContain(`sensitivity ${DEFAULT_CONFIG.mouse.sensitivity}`);
        });

        it('includes volume command', () => {
          const cfg = cfgGenerator(DEFAULT_CONFIG);
          expect(cfg).toContain(`volume ${DEFAULT_CONFIG.audio.masterVolume}`);
        });

        it('includes bind commands for all 15 actions', () => {
          const cfg = cfgGenerator(DEFAULT_CONFIG);
          expect(cfg).toContain('bind "SPACE" "+jump"');
          expect(cfg).toContain('bind "MOUSE1" "+attack"');
        });

        it('outputs voice_enable 0 when muteEnemyVoice is true', () => {
          const cfg = cfgGenerator({
            ...DEFAULT_CONFIG,
            audio: { ...DEFAULT_CONFIG.audio, muteEnemyVoice: true },
          });
          expect(cfg).toContain('voice_enable 0');
        });

        it('outputs voice_enable 1 when muteEnemyVoice is false', () => {
          const cfg = cfgGenerator({
            ...DEFAULT_CONFIG,
            audio: { ...DEFAULT_CONFIG.audio, muteEnemyVoice: false },
          });
          expect(cfg).toContain('voice_enable 1');
        });

        it('returns a non-empty string', () => {
          const cfg = cfgGenerator(DEFAULT_CONFIG);
          expect(typeof cfg).toBe('string');
          expect(cfg.length).toBeGreaterThan(0);
        });
      });
    });
